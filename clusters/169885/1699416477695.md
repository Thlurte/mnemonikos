## if statements

they do something if certain condition is true, else they dont do it at all.
**if**
```cpp
#include <iostream>
int main(){
    int age;

    std::cout << "Enter your age";
    std::cin >> age;

    if(age>=18){
        std::cout << "You are allowed to use the program" << '\n';
    }

    return 0;
}
```
**if & else**
```cpp
#include <iostream>
#include <cmath>
int main(){
    int age;

    std::cout << "Enter your age";
    std::cin >> age;

    if(age>=18){
        std::cout << "You are allowed to use the program" << '\n';
    }
    else {
        std::cout << "Not old enough buddy";
    }

    return 0;
}
```
**else if**
```cpp
#include <iostream>
#include <cmath>
int main(){
    int age;

    std::cout << "Enter your age";
    std::cin >> age;

    if(age>100) {
        std::cout << "ngl you are too old to be human dude..";
    }
    else if(age>=18){
        std::cout << "You are allowed to use the program" << '\n';
    }
    else if(age<0) {
        std::cout << "ngl you are too wrong..";
    }
    else {
        std::cout << "Not old enough buddy";
    }

    return 0;
}
```
