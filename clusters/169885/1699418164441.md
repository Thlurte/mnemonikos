## Ternary operator

is a relpacement for if/else statement
```cpp
#include <iostream>

int main(){
    int age = 15;
    age >= 18 ? std::cout << "Pass" : std::cout << "Fail";
    return 0;
}
```

```cpp
#include <iostream>

int main(){
    int number = 10;
    number % 2 ==0 ? std::cout << "Even" : std::cout << "Odd";
    return 0;
}
```

```cpp
#include <iostream>

int main(){
    int number = 10;
    number % 2 ? std::cout << "Even" : std::cout << "Odd";
    return 0;
}
```

```cpp
#include <iostream>

int main(){
    bool status = true;
    std::cout << (status ? "Hungry" : "Not Hungry");
    return 0;
}
```
