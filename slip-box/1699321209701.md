## Math Functions

```cpp
#include <iostream>
#include <cmath>
int main(){
    double x = 3;
    double y = 5;
    double z;

    z = std::max(x,y);

    std::cout << z << '\n';

    z = std::min(x,y);

    std::cout << z << '\n';

    // cmath functions

    z = pow(2,4);
    std::cout << z << '\n';
    z = sqrt(pow(2,4));
    std::cout << z << '\n';
    z = abs(-x);
    std::cout << z << '\n';
    z = round(3.556);
    std::cout << z << '\n';
    z = round(3.116);
    std::cout << z << '\n';
    z = ceil(3.556);
    std::cout << z << '\n';
    z = ceil(3.116);
    std::cout << z << '\n';
    z = floor(3.556);
    std::cout << z << '\n';
    z = floor(3.116);
    std::cout << z << '\n';

}
```
