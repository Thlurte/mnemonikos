# Const parameters

```cpp
#include <iostream>

void printName(const std::string name,const  int age);

int main(){
    std::string name = "Age";
    int age = 21;
    printName(name,age);
    return 0;
}

void printName(const std::string name,const int age){
    std::cout << name << '\n' << age;
}
```
